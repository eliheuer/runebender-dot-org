<DocSection title="Prerequisites">

- Rust installed from [rust-lang.org](https://rust-lang.org/).
- A current checkout of [runebender-xilem](https://github.com/eliheuer/runebender-xilem).
- A UFO source to open, or the included example UFO from the repo.

<Callout>The Runebender Xilem README currently lists MSRV 1.88. Treat that as a moving project requirement, not a long-term compatibility promise.</Callout>

</DocSection>

<DocSection title="Build">

<CommandList commands={["git clone https://github.com/eliheuer/runebender-xilem.git", "cd runebender-xilem", "cargo run"]} />

</DocSection>

<DocSection title="Open a UFO">

<CommandList
  commands={[
    { cmd: "cargo run", comment: "Opens a file picker" },
    { cmd: "cargo run -- assets/untitled.ufo", comment: "Opens a specific UFO" },
    { cmd: "cargo run -- --verbose", comment: "Runs with verbose logging" },
  ]}
/>

</DocSection>
